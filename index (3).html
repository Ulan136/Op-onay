  <div class="total" id="totalSum">ИТОГО: 0 ₸</div>

  <script>
    let rowCount = 0;

    function addRow(cm = 0, qty = 0) {
      rowCount++;
      const tbody = document.getElementById('tableBody');
      const row = document.createElement('tr');
      row.innerHTML = `
        <td>${rowCount}</td>
        <td><input type="number" value="${cm}" oninput="calculateAll()"></td>
        <td><input type="number" value="${qty}" oninput="calculateAll()"></td>
        <td class="length">0</td>
        <td class="sum">0</td>
        <td><button class="delete-btn" onclick="deleteRow(this)">✕</button></td>
      `;
      tbody.appendChild(row);
      calculateAll();
    }

    function deleteRow(btn) {
      btn.parentElement.parentElement.remove();
      renumberRows();
      calculateAll();
    }

    function renumberRows() {
      const rows = document.querySelectorAll('#tableBody tr');
      rows.forEach((row, i) => {
        row.cells[0].textContent = i + 1;
      });
      rowCount = rows.length;
    }

    function calculateAll() {
      const price = parseFloat(document.getElementById('globalPrice').value) || 0;
      let total = 0;

      document.querySelectorAll('#tableBody tr').forEach(row => {
        const cm = parseFloat(row.cells[1].querySelector('input').value) || 0;
        const qty = parseFloat(row.cells[2].querySelector('input').value) || 0;
        
        const length = cm * qty;
        const sum = Math.round(length * price);

        row.cells[3].textContent = length;
        row.cells[4].textContent = sum.toLocaleString('ru-RU');

        total += sum;
      });

      document.getElementById('totalSum').textContent = `ИТОГО: ${total.toLocaleString('ru-RU')} ₸`;
    }

    function clearAll() {
      if (confirm('Очистить весь список?')) {
        document.getElementById('tableBody').innerHTML = '';
        rowCount = 0;
        document.getElementById('totalSum').textContent = 'ИТОГО: 0 ₸';
      }
    }

    // Запуск
    window.onload = () => {
      for (let i = 0; i < 8; i++) addRow();
    };
